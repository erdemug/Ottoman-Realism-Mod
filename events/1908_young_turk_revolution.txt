#1908 Revolution
youngturkrevolution.1 = {
	type = country_event
	placement = root
	title = "Long Live The Revolution!"
	desc = "Placeholder"
	flavor = "placeholder"

	duration = 3

	event_image = {
		video = "votp_cops_march"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_protest.dds"

	trigger = {
		has_variable = constitution_suspended
		game_date >= 1908.7.1
		has_law = law_type:law_monarchy
		OR = {
			has_law = law_type:law_autocracy
			has_law = law_type:law_oligarchy
		}
		NOT = { has_variable = 1908_revolution_triggered }
	}

	immediate = {
		set_variable = 1908_revolution_triggered
	}

	option = {
		name = "Restore the Constitution! Recall the Parliament"
		default_option = yes
		set_variable = successful_revolution
		activate_law = law_type:law_census_voting
		activate_law = law_type:law_freedom_of_conscience
		activate_law = law_type:law_right_of_assembly
	}

}
