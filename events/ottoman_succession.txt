# Mahmud dies - Abdülmecid is the new Sultan (1839)
ottomansuccession.1 = {
	type = country_event
	placement = root
	title = "Sultan Mahmud is dead"
	desc = "Placeholder"
	flavor = "placeholder"

	duration = 3

	icon = "gfx/interface/icons/event_icons/event_skull.dds"

	gui_window = event_window_1char_tabloid

	left_icon = scope:mahmud_scope

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	
	trigger = {
		has_law = law_type:law_monarchy
		NOT = { has_variable = sultan_mahmud_death }
		game_date >= 1839.7.1
		ruler = {
			has_template = TUR_Mahmud_II
		}		
	}

	immediate = {
		set_variable = sultan_mahmud_death
		ruler = {
			save_scope_as = mahmud_scope
		}
	}

	option = {
		name = "Placeholder"
		default_option = yes
		ruler = {
			kill_character = yes
		}
		heir = {
			set_character_as_ruler = yes
		}
		create_character = {
			template =  TUR_Abdulaziz
			heir = yes
		}

	}
}

# Murad is not well - Abdülhamid II is the new Sultan (1876)
ottomansuccession.4 = {
	type = country_event
	placement = root
	title = "Murad is not well. Who will rule the Empire?"
	desc = "Placeholder"
	flavor = "placeholder"

	duration = 3

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	gui_window = event_window_1char_tabloid
	
	left_icon = scope:abdicating_monarch

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	
	trigger = {
		has_variable = successful_1876_coup
		has_law = law_type:law_monarchy
		NOT = { has_variable = deposed_sultan }
		game_date >= 1876.9.1
		ruler = {
			has_template = TUR_Murad_V
			has_trait = psychological_affliction
		}		
	}

	immediate = {
		set_variable = deposed_sultan
		ruler = {
			save_scope_as = abdicating_monarch
		}
	}

	option = {
		name = "Long live Sultan Abdülhamid-i Sani! He will promulgate the constitution!"
		default_option = yes
		custom_tooltip = {
			text = leader_abdicates_the_throne_tt
			scope:abdicating_monarch = {
				exile_character_with_martyrdom = yes
			}
		}
		heir = {
			set_character_as_ruler = yes
		}
		create_character = {
			template =  TUR_Mehmed_V
			heir = yes
		}
		owner = {
			trigger_event = { id = 1876coup.2 popup = yes }
		}
	}

	# option = {
	# 	name = "He is not that sick though, is he?"
	# }

}
