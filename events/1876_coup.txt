#1876 Coup d'Etat
1876coup.1 = {
	type = country_event
	placement = root
	title = 1876coup.1.t
	desc = 1876coup.1.d
	flavor = 1876coup.1.f

	duration = 3

	event_image = {
		video = "votp_conspiring"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	trigger = {
		has_law = law_type:law_monarchy
		OR = {
			has_law = law_type:law_autocracy
			has_law = law_type:law_oligarchy
		}
		game_date >= 1876.5.30
		NOT = { has_variable = 1876_coup }
		ruler = {
			has_template = TUR_Abdulaziz
		}
	}

	immediate = {
		set_variable = 1876_coup
		create_character = {
			template = agitator_mithat_pasa
			on_created = {
				set_variable = mithat_var
			}
		}
		random_scope_character = {
			limit = {
				has_variable = mithat_var
			}
			save_scope_as = mithat_pasa
		}
	}

	option = {
		name = 1876coup.1.a
		default_option = yes
		set_variable = successful_1876_coup
		owner = {
			trigger_event = { id = ottomansuccession.3 popup = yes }
		}
	}

	# option = {
	# 	name = 1876coup.1.b
	# 	set_variable = defeated_1876_coup
	# }

}

# Suspension of the Constitution

1876coup.3 = {
	type = country_event
	placement = root
	title = 1876coup.3.t
	desc = 1876coup.3.d
	flavor = 1876coup.3.f

	duration = 3

	event_image = {
		video = "europenorthamerica_political_extremism"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	trigger = {
		has_law = law_type:law_monarchy
		NOT = { has_variable = constitution_suspended }
		NOT = { has_variable = questions_about_constitution }
		game_date >= 1878.2.14
		ruler = {
			has_template = TUR_Abdulhamid_II
		}	
		has_variable = constitution_1876
	}

	immediate = {
		set_variable = questions_about_constitution
	}

	option = {
		name = 1876coup.3.a
		default_option = yes
		set_variable = constitution_suspended
		activate_law = law_type:law_autocracy
		activate_law = law_type:law_state_religion
		activate_law = law_type:law_censorship
		add_modifier = {
			name = suppressing_radicals
			months = short_modifier_time
		}
		add_loyalists = {
			strata = rich
			value = very_large_radicals
		}
		add_radicals = {
			value = medium_radicals
			strata = middle
		}
		scope:mithat_pasa = {
			exile_character_with_martyrdom = yes
		}
	}
